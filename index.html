  int *selection_sort(int *arr, int ptrlength) {
    for (int i = 0; i < ptrlength - 1; i++) {
      int min = i;
      for (int j = i + 1; j < ptrlength; j++) {
        if (arr[j] < arr[min]) {
          min = j;
        }
      }
      if (min != i) {
        swap(arr[i], arr[min]);
      }
    }
    return arr;
  }
